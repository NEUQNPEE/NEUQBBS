114 // 检查登录
115 [HttpPost("login")]
116 public IActionResult CheckLogin([FromBody] LoginRequest request)
117 {
118 int userId;
119 // 首先处理自动登录
120 if (AutoLogin(request))
121 {
122 return Ok("自动登录成功");
123 }
124
125 if (string.IsNullOrEmpty(request.UserName) || string.IsNullOrEmpty(request.Password))
126 {
127 logger.LogWarning("前端传入用户名或密码为空的登陆请求");
128 return BadRequest("用户名或密码为空");
129 }
130
131 userId = userBll.CheckLogin(request.UserName, request.Password);
132 if (userId == -1)
133 {
134 logger.LogWarning("用户:{userName}的登录失败", request.UserName);
135 return BadRequest("用户名或密码错误");
136 }
137
138 if (!request.RememberMe)
139 {
140 AddHeader("UserId", userId.ToString());
141 logger.LogInformation("用户:{userName}的登录成功，不记住密码", request.UserName);
142 return Ok("登录成功！");
143 }
144 else
145 {
146 string token = GenerateAutoLoginToken(userId);
147 if (token == string.Empty)
148 {
149 return BadRequest("自动登录Token生成失败");
150 }
151
152 // UserId的键值对
153 Dictionary<string, string> tokens =
154 new() { { "UserId", userId.ToString() }, { "AutoLoginToken", token } };
155
156 AddHeader(tokens);
157
158 logger.LogInformation("用户:{userName}的登录成功，记住密码", request.UserName);
159 return Ok("登录成功,Token生成成功");
160 }
161 }

214 private bool AutoLogin(LoginRequest request)
215 {
216 if (string.IsNullOrEmpty(request.AutoLoginToken))
217 {
218 return false;
219 }
220
221 int userId = userBll.CheckAutoLoginToken(request.AutoLoginToken);
222 if (userId == -1)
223 {
224 // todo 根据用户名获取其id
225 logger.LogWarning("用户:{userName}的自动登录Token无效", request.UserName);
226 return false;
227 }
228
229 AddHeader("UserId", userId.ToString());
230
231 logger.LogInformation("用户:{userName}的自动登录成功", request.UserName);
232 return true;
233 }
234
235 private string GenerateAutoLoginToken(int userId)
236 {
237 string token = userBll.GenerateAutoLoginToken(userId);
238
239 if (string.IsNullOrEmpty(token))
240 {
241 logger.LogWarning("id为{id}的用户Token生成失败", userId);
242 return string.Empty;
243 }
244
245 return token;
246 }

248 private void AddHeader(Dictionary<string, string> tokens)
249 {
250 Response.Headers.Add("Access-Control-Expose-Headers", string.Join(",", tokens.Keys));
251
252 foreach (KeyValuePair<string, string> token in tokens)
253 {
254 Response.Headers.Add(token.Key, token.Value);
255 }
256 }
257
258 private void AddHeader(string key, string value)
259 {
260 Response.Headers.Add("Access-Control-Expose-Headers", key);
261 Response.Headers.Add(key, value);
262 }
