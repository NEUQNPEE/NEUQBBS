73 // 前端传来List<int> mainPostIds,获取所有主贴的最后回复信息
74 [HttpPost("lastreplyinfo/{sectionId}")]
75 public ActionResult<List<LastReplyInfoResponse>> GetAllLastReplyInfoByMainPostIds(int sectionId,[FromBody] List<int> mainPostIds)
76 {
77 if (mainPostIds == null || mainPostIds.Count == 0)
78 {
79 logger.LogWarning("主贴id列表为空，无法获取最后回复信息");
80 return BadRequest("主贴id列表不能为空");
81 }
82
83 // 获取所有主贴的最后回复帖
84 List<Post> lastReplyPosts = new();
85 foreach (int mainPostId in mainPostIds)
86 {
87 Post? lastReplyPost = postBll.GetLastReplyPosts(sectionId, mainPostId);
88 if (lastReplyPost != null)
89 {
90 lastReplyPosts.Add(lastReplyPost);
91 }
92 }
93
94 if (lastReplyPosts.Count == 0)
95 {
96 logger.LogWarning("没有找到板块id({sectionId})下的任何主贴的最后回复", sectionId);
97 return NotFound("没有找到板块id({sectionId})下的任何主贴的最后回复");
98 }
99
100 List<LastReplyInfoResponse> lastReplyInfoResponses = new();
101
102 foreach (Post lastReplyPost in lastReplyPosts)
103 {
104 lastReplyInfoResponses.Add(new LastReplyInfoResponse
105 {
106 MainPostId = lastReplyPost.MainPostId,
107 Username = userBll.GetUserNameById(lastReplyPost.UserId),
108 ReplyTime = lastReplyPost.PublishTime
109 });
110 }
111
112 logger.LogInformation("获取板块id({sectionId})下所有主贴的最后回复信息", sectionId);
113 return Ok(lastReplyInfoResponses);
114 }
115 }
