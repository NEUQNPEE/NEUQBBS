103 [HttpPost("lastreplyinfo/{sectionId}")]
104 public ActionResult<List<LastReplyInfoResponse>> GetAllLastReplyInfoByMainPostIds(
105 int sectionId,
106 [FromBody] List<int> mainPostIds
107 )
108 {
109 if (mainPostIds == null || mainPostIds.Count == 0)
110 {
111 logger.LogWarning("主贴id列表为空，无法获取最后回复信息");
112 return BadRequest("主贴id列表不能为空");
113 }
114
115 // 获取所有主贴的最后回复帖
116 List<Post> lastReplyPosts = new();
117 foreach (int mainPostId in mainPostIds)
118 {
119 Post? lastReplyPost = postBll.GetLastReplyPosts(sectionId, mainPostId);
120 if (lastReplyPost != null)
121 {
122 lastReplyPosts.Add(lastReplyPost);
123 }
124 }
125
126 if (lastReplyPosts.Count == 0)
127 {
128 logger.LogWarning("没有找到板块id({sectionId})下的任何主贴的最后回复", sectionId);
129 return NotFound("没有找到板块id({sectionId})下的任何主贴的最后回复");
130 }
131
132 List<LastReplyInfoResponse> lastReplyInfoResponses = new();
133
134 foreach (Post lastReplyPost in lastReplyPosts)
135 {
136 lastReplyInfoResponses.Add(
137 new LastReplyInfoResponse
138 {
139 MainPostId = lastReplyPost.MainPostId,
140 Username = userBll.GetUserNameById(lastReplyPost.UserId),
141 ReplyTime = lastReplyPost.PublishTime
142 }
143 );
144 }
145
146 logger.LogInformation("获取板块id({sectionId})下所有主贴的最后回复信息", sectionId);
147 return Ok(lastReplyInfoResponses);
148 }
