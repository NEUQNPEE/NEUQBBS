66 {
67 if (response.StatusCode != HttpStatusCode.OK)
68 {
69 State = "Url returns " + response.StatusCode + ", " + response.StatusDescription;
70 return this;

74 State = "Url is not an image";

78 var savePath = Server.MapPath(ServerUrl);
79 if (!Directory.Exists(Path.GetDirectoryName(savePath)))
80 {
81 Directory.CreateDirectory(Path.GetDirectoryName(savePath));
82 }
83 try
84 {
85 var stream = response.GetResponseStream();
86 var reader = new BinaryReader(stream);
87 byte[] bytes;
88 using (var ms = new MemoryStream())
89 {
90 byte[] buffer = new byte[4096];
91 int count;
92 while ((count = reader.Read(buffer, 0, buffer.Length)) != 0)
93 {
94 ms.Write(buffer, 0, count);
95 }
96 bytes = ms.ToArray();
97 }
98 File.WriteAllBytes(savePath, bytes);
99 State = "SUCCESS";
100 }

109 private bool IsExternalIPAddress(string url)
110 {
111 var uri = new Uri(url);
112 switch (uri.HostNameType)
113 {
114 case UriHostNameType.Dns:
115 var ipHostEntry = Dns.GetHostEntry(uri.DnsSafeHost);
116 foreach (IPAddress ipAddress in ipHostEntry.AddressList)
117 {
118 byte[] ipBytes = ipAddress.GetAddressBytes();
119 if (ipAddress.AddressFamily == System.Net.Sockets.AddressFamily.InterNetwork)
120 {
121 if (!IsPrivateIP(ipAddress))
122 {
123 return true;
124 }
125 }
126 }
127 break;
128
129 case UriHostNameType.IPv4:
130 return !IsPrivateIP(IPAddress.Parse(uri.DnsSafeHost));
131 }
132 return false;
133 }

135 private bool IsPrivateIP(IPAddress myIPAddress)
136 {
137 if (IPAddress.IsLoopback(myIPAddress)) return true;
138 if (myIPAddress.AddressFamily == System.Net.Sockets.AddressFamily.InterNetwork)
139 {
140 byte[] ipBytes = myIPAddress.GetAddressBytes();
141 // 10.0.0.0/24
142 if (ipBytes[0] == 10)
143 {
144 return true;
145 }
146 // 172.16.0.0/16
147 else if (ipBytes[0] == 172 && ipBytes[1] == 16)
148 {
149 return true;
150 }
151 // 192.168.0.0/16
152 else if (ipBytes[0] == 192 && ipBytes[1] == 168)
153 {
154 return true;
155 }
156 // 169.254.0.0/16
157 else if (ipBytes[0] == 169 && ipBytes[1] == 254)
158 {
159 return true;
160 }
161 }
162 return false;
